Introduction

Ce cours prolonge les bases (variables, tableaux, objets) en structurant la logique avec des fonctions réutilisables et des branchements conditionnels. Tout le code peut être testé dans la console du navigateur (F12) ou dans un fichier .js.

Préférez === et !== aux comparaisons souples == / != pour éviter les conversions de type surprenantes.

1. Les fonctions

Une fonction est un bloc de code réutilisable qui exécute une tâche précise. Elle peut recevoir des paramètres, les traiter et retourner un résultat.

1.1 — Déclaration (function)

Syntaxe classique. La fonction est hoistée : vous pouvez l’appeler avant sa définition dans le fichier. 

function nomDeLaFonction(parametre1, parametre2) {
  return resultat;
}

function saluer(prenom) {
  return "Bonjour, " + prenom + " !";
}
console.log(saluer("Alice"));
console.log(saluer("Bob"));
Le mot-clé return termine la fonction et renvoie une valeur. Sans return, la valeur retournée est undefined.

Paramètres multiples et valeurs par défaut (ES6+)

function additionner(a, b) {
  return a + b;
}

function presenter(nom, age = 18, ville = "Inconnue") {
  return nom + " a " + age + " ans, habite " + ville;
}
console.log(additionner(5, 3));
console.log(presenter("Luc"));
console.log(presenter("Ana", 25, "Paris"));

1.2 — Expression de fonction

La fonction est assignée à une variable. Elle n’est pas hoistée : définissez-la avant de l’utiliser.

const multiplier = function (a, b) {
  return a * b;
};
const carre = function (n) {
  return n * n;
};

function appliquer(fn, valeur) {
  return fn(valeur);
}
console.log(appliquer(carre, 5));

1.3 — Fonctions fléchées (ES6+)

const diviser = (a, b) => {
  return a / b;
};
const doubler = (n) => n * 2;
const somme = (a, b) => a + b;
const direBonjour = () => "Bonjour le monde !";

1.4 — Portée (scope)

let messageGlobal = "Je suis global";

function exempleScope() {
  let messageLocal = "Je suis local";
  console.log(messageGlobal);
  console.log(messageLocal);
}
exempleScope();
// console.log(messageLocal); // ERREUR hors de la fonction
Limitez les variables globales ; utilisez const et let dans la portée la plus étroite possible.

1.5 — Rest (...) et callbacks

function sommeTotal(...nombres) {
  return nombres.reduce((acc, n) => acc + n, 0);
}

function traiterNombre(nombre, callback) {
  const resultat = callback(nombre);
  console.log("Résultat : " + resultat);
}
traiterNombre(5, (n) => n * 3);

const nombres = [1, 2, 3, 4, 5];
const doubles = nombres.map((n) => n * 2);
console.log(doubles);

2. Les conditions

Les structures conditionnelles exécutent du code différent selon qu’une expression est vraie ou fausse.

2.1 — if, else if, else

function evaluerNote(note) {
  if (note >= 90) return "Excellent — A";
  if (note >= 80) return "Très bien — B";
  if (note >= 70) return "Bien — C";
  if (note >= 60) return "Passable — D";
  return "Insuffisant — F";
}

Opérateurs de comparaison et logiques

Opérateur Signification Exemple Résultat
== Égal (valeur) 5 == "5" true
=== Égal (valeur + type) 5 === "5" false
!= / !== Différent 5 !== "5" true
> < >= <= Ordre 10 > 5 true
&& || ! ET / OU / NON true && false false

2.2 — switch

function nomDuJour(numero) {
  switch (numero) {
    case 1:
      return "Lundi";
    case 2:
      return "Mardi";
    case 7:
      return "Dimanche";
    default:
      return "Numéro invalide (1-7)";
  }
}
Après chaque case, utilisez break ou return, sinon le flux « tombe » dans les case suivants (fall-through). 
function typeDeJour(jour) {
  switch (jour.toLowerCase()) {
    case "samedi":
    case "dimanche":
      return "Week-end";
    case "lundi":
    case "mardi":
    case "mercredi":
    case "jeudi":
    case "vendredi":
      return "Jour de semaine";
    default:
      return "Jour inconnu";
  }
}

2.3 — Opérateur ternaire

const age = 20;
const statut = age >= 18 ? "Majeur" : "Mineur";

const estPair = (n) => (n % 2 === 0 ? "pair" : "impair");

const note = 75;
const mention =
  note >= 90 ? "Excellent" : note >= 70 ? "Bien" : note >= 50 ? "Passable" : "Insuffisant";

Réservez les ternaires imbriqués aux cas vraiment courts ; sinon préférez if/else.

2.4 — Truthy et falsy

Falsy

false, 0, -0, 0n, "", null, undefined, NaN

Truthy

Tout le reste : nombres non nuls, chaînes non vides (y compris "0"), objets, tableaux, fonctions…

function afficherNom(nom) {
  if (nom) {
    console.log("Nom : " + nom);
  } else {
    console.log("Aucun nom fourni");
  }

3. Fonctions et conditions combinées

3.1 — Calcul d’IMC

function calculerIMC(poids, taille) {
  if (poids <= 0 || taille <= 0) {
    return "Valeurs invalides — poids et taille doivent être positifs";
  }
  const imc = poids / (taille * taille);
  const imcArrondi = Math.round(imc * 10) / 10;
  let categorie;
  if (imc < 18.5) categorie = "Insuffisance pondérale";
  else if (imc < 25) categorie = "Poids normal";
  else if (imc < 30) categorie = "Surpoids";
  else categorie = "Obésité";
  return `IMC : ${imcArrondi} — ${categorie}`;
}

3.2 — Connexion simple

const utilisateurs = [
  { login: "admin", motDePasse: "Admin123!" },
  { login: "alice", motDePasse: "Alice456#" },
  { login: "bob", motDePasse: "Bob789@" },
];

function seConnecter(login, motDePasse) {
  if (!login || !motDePasse) {
    return "Erreur : login et mot de passe requis";
  }
  const user = utilisateurs.find(
    (u) => u.login === login && u.motDePasse === motDePasse
  );
  return user
    ? `Connexion réussie ! Bienvenue, ${user.login}`
    : "Erreur : identifiants incorrects";
}

3.3 — Jeu de devinette (closures)

const creerJeu = (nombreSecret, maxEssais = 5) => {
  let essaisRestants = maxEssais;
  const deviner = (nombre) => {
    if (essaisRestants <= 0) {
      return "Partie terminée — plus d'essais disponibles";
    }
    essaisRestants--;
    if (nombre === nombreSecret) {
      return `Bravo ! Vous avez trouvé ${nombreSecret} !`;
    }
    if (nombre < nombreSecret) {
      return `Trop petit ! Il vous reste ${essaisRestants} essai(s)`;
    }
    return `Trop grand ! Il vous reste ${essaisRestants} essai(s)`;
  };
  return { deviner };
};

5. Solutions (extraits)

Exercice 1

const celsiusVersFahrenheit = (c) => Math.round(((c * 9) / 5 + 32) * 10) / 10;
const fahrenheitVersCelsius = (f) => Math.round((((f - 32) * 5) / 9) * 10) / 10;

function convertir(valeur, unite) {
  switch (unite.toUpperCase()) {
    case "C":
      return celsiusVersFahrenheit(valeur) + "°F";
    case "F":
      return fahrenheitVersCelsius(valeur) + "°C";
    default:
      return "Unité inconnue — utilisez C ou F";
  }
}

Exercice 2

function verifierMotDePasse(mdp) {
  if (!mdp || mdp.length < 8) {
    return { valide: false, message: "Trop court (min. 8 caractères)" };
  }
  if (!/[A-Z]/.test(mdp)) return { valide: false, message: "Manque une majuscule" };
  if (!/[0-9]/.test(mdp)) return { valide: false, message: "Manque un chiffre" };
  if (!/[!@#$%]/.test(mdp)) {
    return { valide: false, message: "Manque un caractère spécial (!@#$%)" };
  }
  return { valide: true, message: "Mot de passe valide !" };
}

Exercice 3

function calculer(a, operateur, b) {
  switch (operateur) {
    case "+":
      return a + b;
    case "-":
      return a - b;
    case "*":
      return a * b;
    case "%":
      return a % b;
    case "**":
      return Math.pow(a, b);
    case "/":
      if (b === 0) return "Erreur : division par zéro";
      return a / b;
    default:
      return "Opérateur inconnu : " + operateur;
  }
}
